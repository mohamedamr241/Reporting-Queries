CASE 
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 1 THEN 'January'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 2 THEN 'February'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 3 THEN 'March'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 4 THEN 'April'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 5 THEN 'May'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 6 THEN 'June'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 7 THEN 'July'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 8 THEN 'August'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 9 THEN 'September'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 10 THEN 'October'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 11 THEN 'November'
        WHEN EXTRACT(MONTH FROM CREATEDTIME) = 12 THEN 'December'
    END
